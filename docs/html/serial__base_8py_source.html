85  def _thrd_llrecv_from_connection_interface(self):
86  """
87  Receive and process data from serial connection in low-level.
88  Returns:
89  None.
90  """
91  _mident = '%s.%s()' % (self.__class__.__name__, currentframe().f_code.co_name)
92  BuiltIn().log("%s: lowlevel receiver thread started." % _mident, constants.LOG_LEVEL_INFO)
93  while self._is_connected is False and not self._llrecv_thrd_term.isSet():
94  time.sleep(ConnectionBase.RECV_MSGS_POLLING_INTERVAL)
95 
96  while not self._llrecv_thrd_term.isSet():
97  # noinspection PyBroadException
98  # implementation from here:
99  # http://sourceforge.net/p/pyserial/code/HEAD/tree/trunk/pyserial/examples/rfc2217_server.py
100  try:
101  data = self.socket.read(1) # read one, blocking
102  n = self.socket.inWaiting() # look if there is more
103  if n:
104  read_data = self.socket.read(n)
105  data = data.decode('utf-8') + read_data.decode('utf-8') # and get as much as possible
106  for character in data:
107  self.serial_queue.put(character)
108  except Exception as _reason:
109  # ignore all errors.
110  pass
111 
112  time.sleep(ConnectionBase.RECV_MSGS_POLLING_INTERVAL)
113 
114  # _read must have chance to terminate, too,therefore
115  # wait here 5 times polling interval
116  time.sleep(ConnectionBase.RECV_MSGS_POLLING_INTERVAL * 5)
117 
118  self._llrecv_thrd_term.clear()
119  BuiltIn().log("%s: lowlevel receiver thread terminated." % _mident, constants.LOG_LEVEL_INFO)

163  def _read(self):
164  """
165  Receive data from serial connection.
166  Returns:
167  Data received from connection.
168  """
169  data = ''
170  # usually \r\n or \n is sent to terminate a line,
171  # but U-Boot sends \n\r, therefore try to fit to all
172  # possible line endings and return the identified line.
173 
174  # We read the data here from the queue filled by the
175  # low-level receiver thread.
176  while (data[-1:] != '\n') and not self._llrecv_thrd_term.isSet():
177  try:
178  d = self.serial_queue.get(block=False)
179  data = data + d
180  except queue.Empty:
181  # non blocking get from serq causes that
182  # Queue.Empty exception. In this case wait some milliseconds before
183  # retry
184  time.sleep(ConnectionBase.RECV_MSGS_POLLING_INTERVAL)
185 
186  # remove the \n
187  data = data[:-1]
188 
189  # if we filter for \n, then
190  # if a \r\n was sent, we need to remove the remaining \r
191  if data[-1:] == '\r':
192  data = data[:-1]
193  # if \n\r was sent, then the next line starts with a \r which
194  # need to be removed.
195  # Disadvantage: If somebody uses \r to overwrite an existing line,
196  # this will not work because the in this case intended \r will be filtered,
197  # too.
198  if data[:1] == '\r':
199  data = data[1:]
200 
201  # SerialSocket decodes all characters from UTF-8 to unicode with mode "replace".
202  # This avoids that data waste caused e.g. by startup/shutdown of the target can bring corrupt data into
203  # the TML Framework.
204  return self._q_dollar(data)

208  """
209  Quit serial connection.
210  Returns:
211  None
212  """
213  self._is_connected = False
214  if self._llrecv_thrd_obj and self._llrecv_thrd_obj.is_alive():
215  self._llrecv_thrd_term.set()

217  time.sleep(ConnectionBase.RECV_MSGS_POLLING_INTERVAL)
218  self._llrecv_thrd_obj = None
