147  elif self._authentication == AuthenticationType.PASSWORDKEYFILE:

150  # From http://docs.paramiko.org/en/1.13/api/client.html
151  # Authentication is attempted in the following order of priority:
152  # 1. The pkey or key_filename passed in (if any)
153  # 2. Any key we can find through an SSH agent

212  def _read(self):
213  """
214  Read data from SSH connection.
215  Returns:
216  Data from SSH connection.
217  """
218  data = ''
219 
220  # We read the data here from the queue filled by the
221  # low-level receiver thread.
222  while data[-2:] != '\r\n' and not self._llrecv_thrd_term.isSet():
223  try:
224  d = self.SSHq.get(block=False)
225 
226  if data[-2:] != '\r\n':
227  data = data + d
228  except queue.Empty:
229  # non blocking get from SSHq causes that
230  # Queue.Empty exception. In this case wait some milliseconds before
231  # retry
232  time.sleep(TCPBase.RECV_MSGS_POLLING_INTERVAL)
233 
234  data = data[:-2]
