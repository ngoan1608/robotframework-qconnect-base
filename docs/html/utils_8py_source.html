28 class Singleton(object): # pylint: disable=R0903

125  @staticmethod
126  def caller_name(skip=2):
127  """
128  Get a name of a caller in the format module.class.method
129  Args:
130  skip: specifies how many levels of stack to skip while getting caller
131  name. skip=1 means "who calls me", skip=2 "who calls my caller" etc.
132 
133  Returns:
134  An empty string is returned if skipped levels exceed stack height
135  """
136  stack = inspect.stack()
137  start = 0 + skip
138  if len(stack) < start + 1:
139  return ''
140  parent_frame = stack[start][0]
141 
142  name = []
143  module = inspect.getmodule(parent_frame)
144  # `modname` can be None when frame is executed directly in console
145  # TODO(techtonik): consider using __main__
146  if module:
147  name.append(module.__name__)
148  # detect classname
149  if 'self' in parent_frame.f_locals:
150  # I don't know any way to detect call from the object method
151  # XXX: there seems to be no way to detect static method call - it will
152  # be just a function call
153  name.append(parent_frame.f_locals['self'].__class__.__name__)
154  codename = parent_frame.f_code.co_name
155  if codename != '<module>': # top level usually
156  name.append(codename) # function or a method
157  del parent_frame
158  return ".".join(name)
