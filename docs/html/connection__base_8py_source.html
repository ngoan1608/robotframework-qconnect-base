212  """

214  Args:
215  sync_with_start: determine if thread needs to wait for start event.
216 
217  Returns:
218  None
219  """
220  _mident = '%s.%s()' % (self.__class__.__name__, currentframe().f_code.co_name)

226  BuiltIn().log("%s: receiver thread started." % _mident, constants.LOG_LEVEL_DEBUG)
227  while not self._recv_thrd_term.isSet():

229  msg = self.read_obj()
230  if self._should_check_timeout:
231  self.check_timeout(msg)

235  self.pre_msg_check(msg)
236  BuiltIn().log(msg, constants.LOG_LEVEL_INFO)

243  if self.__class__._traceq_obj:
244  for (regex_filter, msg_queue, back_trace_queue, use_fetch_block, regex_end_block_pattern, regex_line_filter) in self.__class__._traceq_obj.values():
245  is_hit = False
246  result_obj = None
247  if use_fetch_block is True:
248  matchObj = regex_line_filter.search(msg)

318  def wait_4_trace(self, search_obj, timeout=0, use_fetch_block=False, end_of_block_pattern=".*", filter_pattern=".*", *fct_args):
319  """
320  Suspend the control flow until a Trace message is received which matches to a specified regular expression.
321 
322  Args:
323  search_obj : Regular expression all received trace messages are compare to.
324  Can be passed either as a string or a regular expression object. Refer to Python documentation for module 're'.
325  use_fetch_block : Determine if 'fetch block' feature is used.
326  end_of_block_pattern : The end of block pattern.
327  filter_pattern : Regular expression object to filter message line by line.
328  timeout : Optional timeout parameter specified as a floating point number in the unit 'seconds'.
329  fct_args: Optional list of function arguments passed to be sent.
330 
331  Returns:
332  None : If no trace message matched to the specified regular expression and a timeout occurred.
333  <match> : If a trace message has matched to the specified regular expression, a match object is returned as the result.
334  The complete trace message can be accessed by the 'string' attribute of the match object.
335  For access to groups within the regular expression, use the group() method.
336  For more information, refer to Python documentation for module 're'.
337 
338  Disabled pyLint Messages:
339  W0703: Catch 'Exception' Used when an except catches Exception instances.
340 
341  There is no specific exception to be handled and script should continue in any case.
342  This method calls a function object passes as an argument. So it has no knowledge about possible exception classes.
343  """
344  _mident = '%s.%s()' % (self.__class__.__name__, currentframe().f_code.co_name)
345  BuiltIn().log('Execute %s' % _mident, constants.LOG_LEVEL_DEBUG)
346  search_regex = re.compile(search_obj, re.M | re.S | re.U)
347  regex_obj_filter = re.compile(filter_pattern)
348  trq_handle, trace_queue = self.create_and_activate_trace_queue(search_regex, use_fetch_block, end_of_block_pattern, regex_obj_filter)
349 
350  try:
351  self.send_obj(*fct_args)
352  except Exception as err_msg: # pylint: disable=W0703
353  BuiltIn().log('%s: An Exception occurred executing function object: %s' % (_mident, repr(self.send_obj)), 'ERROR')
354  BuiltIn().log('Function Arguments: %s' % repr(fct_args), 'ERROR')
355  BuiltIn().log('Error Message: %s' % repr(err_msg), 'ERROR')
356  success = True
357  match = None
358  try:
359  (dummy, match) = trace_queue.get(True, timeout)
360  except queue.Empty:
361  success = False
362  finally:
363  self.deactivate_and_delete_trace_queue(trq_handle, trace_queue)
364 
365  BuiltIn().log('Completed %s' % _mident, constants.LOG_LEVEL_DEBUG)
366  if success:
